@startuml
title Процесс "Рабочее место исполнителя"

start

partition "Исполнитель" {
  :Авторизоваться в системе;
}

partition "Система" {
  :Проверить права доступа;
  :Загрузить список назначенных\nэтапов/задач;
  :Показать список\nназначенных этапов;
}

partition "Исполнитель" {
  :Просмотреть список этапов;
  :Выбрать этап для работы;
}

partition "Система" {
  :Показать детали этапа\n(описание, сроки,\nресурсы, статус);
}

partition "Исполнитель" {
  if (Начать работу по этапу?) then (Да)
    :Нажать "Старт этапа";
  else (Нет)
    :Выбрать другой этап\nили выйти из системы;
    stop
  endif
}

partition "Система" {
  :Зафиксировать время\nначала работ по этапу;
  :Перевести этап в статус\n"В работе" (если применимо);
}

' Основной цикл работы по этапу
while (Этап не завершен?) is (Да)
  partition "Исполнитель" {
    if (Нужны дополнительные\nресурсы?) then (Да)
      :Сформировать запрос\nдополнительных ресурсов;
      :Отправить запрос;
      partition "Система" {
        :Зарегистрировать запрос\nресурсов по этапу;
        :Уведомить Работника склада\n/ Руководителя (по правилам);
      }
      partition "Исполнитель" {
        :Ожидать подтверждения/\nвыдачи ресурсов (если требуется);
      }
    endif

    if (Зафиксировать\nпромежуточный прогресс?) then (Да)
      :Ввести процент выполнения,\nкомментарий,\nключевые метки времени;
      partition "Система" {
        :Сохранить данные\nо прогрессе;
      }
    endif

    if (Работы по этапу завершены?) then (Да)
      :Ввести комментарий/\nотчет по выполнению этапа;
      :Нажать "Завершить этап";
      partition "Система" {
        :Зафиксировать время\nзавершения этапа;
        :Перевести этап в статус\n"Завершен";
      }
    endif
  }
endwhile (Нет)

partition "Исполнитель" {
  :Вернуться к списку этапов\nили выйти из системы;
}

stop
@enduml
